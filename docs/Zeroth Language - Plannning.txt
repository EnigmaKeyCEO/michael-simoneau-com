Minimal VM Logic (Phase-2-Lite)
In zeroth/vm/vm.py (simplified starter):
class ZerothVM:
    def __init__(self, zir):
        self.states = zir["states"]
        self.values = {s: 0.0 for s in self.states}
        self.iteration = 0


    def step(self):
        # trivial oscillation just to prove life
        for k in self.values:
            self.values[k] += 0.01
        self.iteration += 1


    def projection(self):
        return " ".join(self.states)

Minimal Web Server
In zeroth/web/server.py:
from http.server import BaseHTTPRequestHandler, HTTPServer
import threading
import time


def serve(vm):
    def loop():
        while True:
            vm.step()
            time.sleep(0.1)


    threading.Thread(target=loop, daemon=True).start()


    class Handler(BaseHTTPRequestHandler):
        def do_GET(self):
            body = f"""
            <html>
              <body>
                <h1>{vm.projection()}</h1>
                <p>Iteration: {vm.iteration}</p>
              </body>
            </html>
            """
            self.send_response(200)
            self.send_header("Content-Type", "text/html")
            self.end_headers()
            self.wfile.write(body.encode())


    HTTPServer(("localhost", 101), Handler).serve_forever()
